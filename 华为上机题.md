### 问题1、求最远距离

输入坐标数据格式（x,y） 0<x,y,<1000，存在非法坐标（01，1），（0，100）等

总部坐标为(0,0) 某位置相对总部距离为x * x + y * y

输入：ferage12s(100,200)f2dfs(300,400)

输出：(300,400)

自己的解法1

```
bool CheckStr(const string& str) {
    if (str.empty())
        return false;
    for (int i = 0; i < str.size(); ++i) {
        if (isalnum(str[i]) || str[i] == ','
        || str[i] == '(' || str[i] == ')')
            continue;
        return false;
    }
    return true;
}

int main() {
    string str;

    int nPos[10][2] = {};
    while (getline(cin, str)) {
        if (!CheckStr(str)) {
            printf("格式错误！");
            return 0;
        }
        int nCount = 0;
        for (int i = 0; i < str.size(); ++i) {
            string temp;
            if (str[i] == '(') {
                ++i;
                while (str[i] != ',') {
                    if (!isdigit(str[i])) {
                        printf("格式错误！");
                        return 0;
                    }
                    temp.push_back(str[i]);
                    ++i;
                }

                nPos[nCount][0] = atoi(temp.c_str());
                temp.clear();
                ++i;
                while (str[i] != ')') {
                    if (!isdigit(str[i])) {
                        printf("格式错误！");
                        return 0;
                    }
                    temp.push_back(str[i]);
                    ++i;
                }
                nPos[nCount][1] = atoi(temp.c_str());
                temp.clear();
                ++nCount;
            }
        }

        int nMaxDistance = 0;
        int curDistance = 0;
        int nIndex = 0;
        for (int i = 0; i < 10; ++i) {
            if (nPos[i][0] != 0 && nPos[i][1] != 0 )
                curDistance = nPos[i][0] * nPos[i][0] + nPos[i][1] * nPos[i][1];
            if (curDistance > nMaxDistance) {
                nMaxDistance = curDistance;
                nIndex = i;
            }
        }
        printf("(%d, %d) \n", nPos[nIndex][0], nPos[nIndex][1]);
    }
}
```

### 问题2、排序输出磁盘容量

1024M=1G

1024G=1T

```
输入:
3
1G
2G
1024M

输出：
1G
1024M
2G
```

```
输入:
3
2G4M
3M2G
1T

输出：
3M2G
2G4M
1T
```



自己的代码

```
#include <iostream>
#include <map>
using namespace std;

bool CheckFormat(const string& str) {
    if (str.empty())
        return false;
    for (int i = 0; i < str.size(); ++i) {
        if (isdigit(str[i]) || str[i] == 'G'
        || str[i] == 'T' || str[i] == 'M' )
            continue;
        return false;
    }
    return true;
}

int ConvertToM(const string& str) {
    string temp;
    int num = 0;
    for (int i = 0; i < str.size(); ++i) {
        if (str[i] != 'G' && str[i] != 'T' && str[i] != 'M')
            temp.push_back(str[i]);
        if (str[i] == 'M') {
            num += atoi(temp.c_str());
            temp.clear();
        } else if (str[i] == 'G') {
            num += atoi(temp.c_str()) * 1024;
            temp.clear();
        } else if (str[i] == 'T') {
            num += atoi(temp.c_str()) * 1024 * 1024;
            temp.clear();
        }
    }
    return num;
}

int Compare(const void* num1, const void* num2) {
    return *(int*)num1 - *(int*)num2;
}

int main() {
    string num;
    cin >> num;
    for (int i = 0; i < num.size(); ++i) {
        if (!isdigit(num[i])) {
            printf("格式错误！");
            return 0;
        }
    }

    int nCount = atoi(num.c_str());
    string strData[nCount];
    int nData[nCount];

    map<string, int> dic;
    for (int i = 0; i < nCount; ++i) {
        cin >> strData[i];
        if (!CheckFormat(strData[i])) {
            printf("数据格式错误！");
            return 0;
        }
        nData[i] = ConvertToM(strData[i]);
        dic[strData[i]] = nData[i];
    }

    qsort(nData, nCount, sizeof(int), Compare);

    for (int i = 0; i < nCount; ++i) {
        for (auto ite = dic.begin(); ite != dic.end(); ++ite) {
            if (ite->second == nData[i]) {
                cout << ite->first << endl;
                ite->second = 0;
            }
        }
    }
    return 0;
}
```
