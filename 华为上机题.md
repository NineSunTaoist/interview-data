## 7、华为上机题

### 问题1、求最远距离

输入坐标数据格式（x,y） 0<x,y,<1000，存在非法坐标（01，1），（0，100）等

总部坐标为(0,0) 某位置相对总部距离为x * x + y * y

输入：ferage12s(100,200)f2dfs(300,400)

输出：(300,400)

自己的解法1

```
bool CheckStr(const string& str) {
    for (int i = 0; i < str.size(); ++i) {
        if (isalnum(str[i]) || str[i] == ','
        || str[i] == '(' || str[i] == ')')
            continue;
        return false;
    }
    return true;
}

int main() {
    string str;

    int nPos[10][2] = {};
    while (getline(cin, str)) {
        if (!CheckStr(str)) {
            printf("格式错误！");
            return 0;
        }
        int nCount = 0;
        for (int i = 0; i < str.size(); ++i) {
            string temp;
            if (str[i] == '(') {
                ++i;
                while (str[i] != ',') {
                    if (!isdigit(str[i])) {
                        printf("格式错误！");
                        return 0;
                    }
                    temp.push_back(str[i]);
                    ++i;
                }

                nPos[nCount][0] = atoi(temp.c_str());
                temp.clear();
                ++i;
                while (str[i] != ')') {
                    if (!isdigit(str[i])) {
                        printf("格式错误！");
                        return 0;
                    }
                    temp.push_back(str[i]);
                    ++i;
                }
                nPos[nCount][1] = atoi(temp.c_str());
                temp.clear();
                ++nCount;
            }
        }

        int nMaxDistance = 0;
        int curDistance = 0;
        int nIndex = 0;
        for (int i = 0; i < 10; ++i) {
            if (nPos[i][0] != 0 && nPos[i][1] != 0 )
                curDistance = nPos[i][0] * nPos[i][0] + nPos[i][1] * nPos[i][1];
            if (curDistance > nMaxDistance) {
                nMaxDistance = curDistance;
                nIndex = i;
            }
        }
        printf("(%d, %d) \n", nPos[nIndex][0], nPos[nIndex][1]);
    }
}
```
